#include <stdio.h>

int main() {
    int vetor[5];
    int count = 5;
    int maior;
    int len = sizeof(vetor)/sizeof(int);
    printf("Digite 5 numeros: ");
    for (int i=0; i<count; i++){
        scanf("%i", &vetor[i]);
    }
    for(int i=0; i<len; i++){
        if(vetor[i] > maior){
            maior = vetor[i];
        }
    }
    printf("Maior: %i", maior);
    return 0;
}

#include <stdio.h>    
     
int main()    
{      
    int vetor[5];     
    int count = 5;
    int troca = 0;    
    int len = sizeof(vetor)/sizeof(int);
    printf("Digite 5 numeros: ");
    for (int i=0; i<count; i++){
        scanf("%i", &vetor[i]);
    }
    for (int i = 0; i < len; i++) {     
        for (int j = i+1; j < len; j++) {     
           if(vetor[i] > vetor[j]) {    
               troca = vetor[i];
               vetor[i] = vetor[j];
               vetor[j] = troca;
           }     
        }   
    }
    
    for(int i = 0; i < len; i++){
        printf(" %i", vetor[i]);
    }
    
    return 0;    
}

#include <stdio.h>    
     
int main()    
{      
    int vetor[5];     
    int count = 5;
    int numero;
    int soma;
    int len = sizeof(vetor)/sizeof(int);
    printf("Digite 5 numeros: ");
    for (int i=0; i<count; i++){
        scanf("%i", &vetor[i]);
    }
    
    printf("Selecione o numero repetido:");
    scanf("%i",&numero);
    for (int i = 0; i < len; i++) {     
        if(vetor[i] == numero){
            soma++;
        }
    }
    
    printf("%i", soma);
    
    return 0;    
}
